266 private IEnumerator RunAudioJob(AudioJob job)
267 {
268 //Play the delay that has been set to the job
269 yield return new WaitForSeconds(job.delay);
270
271 var key = new Tuple<AudioType, GameObject>(job.type, job.requester);
272 AudioTrack track = (AudioTrack) _audioTable[key];
273 track.source.clip = GetAudioClipFromAudioTrack(job.type, track);
274
275 switch (job.action)
276 {
277 case AudioAction.START:
278 track.source.Play();
279 break;
280
281 case AudioAction.STOP:
282 //The audio can not be stopped immediately. There would be no fading in that case
283 if(job.fade == 0) track.source.Stop();
284 break;
285
286 case AudioAction.RESTART:
287 track.source.Stop();
288 track.source.Play();
289 break;
290 }
291
292 if (job.fade > 0)
293 {
294 float initialVolume = job.action is AudioAction.START or AudioAction.RESTART ? 0 : track.source.volume;
295 float targetVolume = initialVolume == 0 ? track.source.volume : 0;
296
297 float timeElapsed = 0f;
298
299 while (timeElapsed <= job.fade)
300 {
301 track.source.volume = Mathf.Lerp(initialVolume, targetVolume, timeElapsed / job.fade);
302 timeElapsed += Time.deltaTime;
303 yield return null;
304 }
305
306 if (job.action == AudioAction.STOP)
307 {
308 track.source.Stop();
309 track.source.volume = initialVolume; //In case I want to restart the sound with no fading
310 }
311 }
312
313 _jobTable.Remove(new Tuple<AudioType, GameObject>(job.type, job.requester));
314 Log("Job count: "+_jobTable.Count);
315
316 //yield return null;
317 }
